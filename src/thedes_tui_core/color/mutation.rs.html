64#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Default)]
65pub struct MutateBrightness<M>(pub M);
66
67impl<M> Mutation<Color> for MutateBrightness<M>
68where
69    M: Mutation<Brightness>,
70{
71    fn mutate(
72        self,
73        mut target: Color,
74    ) -> Result<Color, <Color as Mutable>::Error> {
75        let Self(mutation) = self;
76        let brightness = target.approx_brightness()?;
77        target.set_approx_brightness(mutation.mutate(brightness)?)?;
78        Ok(target)
79    }
80}
