3pub(crate) trait ColorToCrossterm {
4    fn to_crossterm(&self) -> crossterm::style::Color;
5}
6
7impl ColorToCrossterm for Color {
8    fn to_crossterm(&self) -> crossterm::style::Color {
9        match self {
10            Self::Basic(color) => color.to_crossterm(),
11            Self::LegacyRgb(color) => color.to_crossterm(),
12            Self::Rgb(color) => color.to_crossterm(),
13            Self::Grayscale(color) => color.to_crossterm(),
14        }
15    }
16}
17
18impl ColorToCrossterm for BasicColor {
19    fn to_crossterm(&self) -> crossterm::style::Color {
20        match self {
21            Self::Black => crossterm::style::Color::Black,
22            Self::DarkRed => crossterm::style::Color::DarkRed,
23            Self::DarkGreen => crossterm::style::Color::DarkGreen,
24            Self::DarkYellow => crossterm::style::Color::DarkYellow,
25            Self::DarkBlue => crossterm::style::Color::DarkBlue,
26            Self::DarkMagenta => crossterm::style::Color::DarkMagenta,
27            Self::DarkCyan => crossterm::style::Color::DarkCyan,
28            Self::LightGray => crossterm::style::Color::Grey,
29            Self::DarkGray => crossterm::style::Color::DarkGrey,
30            Self::LightRed => crossterm::style::Color::Red,
31            Self::LightGreen => crossterm::style::Color::Green,
32            Self::LightYellow => crossterm::style::Color::Yellow,
33            Self::LightBlue => crossterm::style::Color::Blue,
34            Self::LightMagenta => crossterm::style::Color::Magenta,
35            Self::LightCyan => crossterm::style::Color::Cyan,
36            Self::White => crossterm::style::Color::White,
37        }
38    }
39}
40
41impl ColorToCrossterm for LegacyRgb {
42    fn to_crossterm(&self) -> crossterm::style::Color {
43        crossterm::style::Color::AnsiValue(self.code())
44    }
45}
46
47impl ColorToCrossterm for Rgb {

52            b: self.blue,

54    }
55}
