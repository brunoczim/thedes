133            Self::DarkCyan => (BasicColorVariant::Dark, BasicColorCore::Cyan),
134            Self::LightGray => (BasicColorVariant::Dark, BasicColorCore::White),
135            Self::DarkGray => (BasicColorVariant::Light, BasicColorCore::Black),
136            Self::LightRed => (BasicColorVariant::Light, BasicColorCore::Red),
137            Self::LightGreen => {
138                (BasicColorVariant::Light, BasicColorCore::Green)

155impl TryFrom<u8> for BasicColor {

159        let decoded = match code {
160            Self::BLACK_CODE => Self::Black,
161            Self::DARK_RED_CODE => Self::DarkRed,
162            Self::DARK_GREEN_CODE => Self::DarkGreen,
163            Self::DARK_YELLOW_CODE => Self::DarkYellow,
164            Self::DARK_BLUE_CODE => Self::DarkBlue,
165            Self::DARK_MAGENTA_CODE => Self::DarkMagenta,
166            Self::DARK_CYAN_CODE => Self::DarkCyan,
167            Self::LIGHT_GRAY_CODE => Self::LightGray,
168            Self::DARK_GRAY_CODE => Self::DarkGray,
169            Self::LIGHT_RED_CODE => Self::LightRed,
170            Self::LIGHT_GREEN_CODE => Self::LightGreen,
171            Self::LIGHT_YELLOW_CODE => Self::LightYellow,
172            Self::LIGHT_BLUE_CODE => Self::LightBlue,
173            Self::LIGHT_MAGENTA_CODE => Self::LightMagenta,
174            Self::LIGHT_CYAN_CODE => Self::LightCyan,
175            Self::WHITE_CODE => Self::White,
176            _ => Err(BadBassicColorCode(code))?,
177        };
178        Ok(decoded)
179    }

182impl ApproxBrightness for BasicColor {
183    fn approx_brightness(&self) -> Result<Brightness, BrightnessError> {
184        let red_weight = RED_MILLI_WEIGHT;
185        let green_weight = GREEN_MILLI_WEIGHT;
186        let blue_weight = BLUE_MILLI_WEIGHT;
187        let total_base = red_weight + green_weight + blue_weight;
188        let soft_max = total_base * 2 - 1;
189
190        let parts = self.decode_parts();
191        let base = match parts.core {
192            BasicColorCore::Black => 0,
193            BasicColorCore::Red => red_weight * 2,
194            BasicColorCore::Green => green_weight * 2,
195            BasicColorCore::Yellow => (red_weight + green_weight) / 2,
196            BasicColorCore::Blue => blue_weight * 2,
197            BasicColorCore::Magenta => (red_weight + blue_weight) / 2,
198            BasicColorCore::Cyan => (blue_weight + green_weight) / 2,
199            BasicColorCore::White => soft_max * 3 / 4,
200        };
201
202        let light_bonus = match parts.variant {
203            BasicColorVariant::Dark => 0,
204            BasicColorVariant::Light => total_base,
205        };
206
207        let in_basic_color_space = (base + light_bonus).min(soft_max);
208
209        Brightness::new(in_basic_color_space).spread_level(soft_max)
210    }
211}

213impl MutableApproxBrightness for BasicColor {
214    fn set_approx_brightness(
215        &mut self,
216        brightness: Brightness,
217    ) -> Result<(), BrightnessError> {
218        let parts = self.decode_parts();
219
220        let light_variant = Self::encode_parts(BasicColorParts {
221            variant: BasicColorVariant::Light,
222            ..parts
223        });
224
225        let dark_variant = Self::encode_parts(BasicColorParts {
226            variant: BasicColorVariant::Dark,
227            ..parts
228        });
229
230        let light_brightness = light_variant.approx_brightness()?;
231
232        *self = if brightness >= light_brightness {
233            light_variant
234        } else {
235            dark_variant
236        };
237
238        Ok(())
239    }
240}
