129#[derive(Debug, Clone)]

131    view: Rect,
132    config: Config,

134
135impl Camera {
136    fn new(config: Config) -> Self {
137        Self {
138            view: Rect {
139                top_left: CoordPair::from_axes(|_| 0),
140                size: CoordPair::from_axes(|_| 0),
141            },
142            config,
143        }
144    }
145
146    fn update(&mut self, available_canvas: CoordPair, game: &Game) {
147        if self.view.size != available_canvas {
148            self.center_on_player(available_canvas, game);
149        } else if !self
150            .freedom_view()
151            .contains_point(game.player().position().head())
152        {
153            self.stick_to_border(game);
154        } else if !self.view.contains_point(game.player().position().head())
155            || !self.view.contains_point(game.player().position().pointer())
156        {
157            self.center_on_player(available_canvas, game);
158        }
159    }
160
161    pub fn render(
162        &mut self,
163        app: &mut App,
164        game: &Game,
165        dynamic_style: &DynamicStyle,
166    ) -> Result<(), Error> {
167        let available_canvas =
168            self.available_canvas_size(app, dynamic_style)?;
169
170        self.update(available_canvas, game);
171
172        app.canvas
173            .queue([screen::Command::new_clear_screen(BasicColor::Black)]);
174
175        let pos_string = format!("↱{}", game.player().position().head());
176        text::styled(app, &pos_string, &text::Style::default())?;
177
178        for y in self.view.top_left.y .. self.view.bottom_right().y {
179            for x in self.view.top_left.x .. self.view.bottom_right().x {
180                let player_pos = game.player().position();
181                let point = CoordPair { y, x };
182                let canvas_point =
183                    point - self.view.top_left + dynamic_style.margin_top_left;
184
185                let ground = game.map().get_ground(point)?;
186                let bg_color = match ground {
187                    Ground::Grass => Rgb::new(0x00, 0xff, 0x80).into(),
188                    Ground::Sand => Rgb::new(0xff, 0xff, 0x80).into(),
189                    Ground::Stone => Rgb::new(0xc0, 0xc0, 0xc0).into(),
190                };
191
192                let block = game.map().get_block(point)?;
193
194                let fg_color = BasicColor::Black.into();
195                let char = match block {
196                    Block::Special(SpecialBlock::Player) => {
197                        if player_pos.head() == point {
198                            'O'
199                        } else {
200                            match player_pos.facing() {
201                                Direction::Up => 'Ʌ',
202                                Direction::Down => 'V',
203                                Direction::Left => '<',
204                                Direction::Right => '>',
205                            }
206                        }
207                    },
208                    Block::Special(SpecialBlock::Monster(id)) => {
209                        let monster_pos =
210                            game.monster_registry().get_by_id(id)?.position();
211                        match monster_pos.facing() {
212                            Direction::Up => 'ɷ',
213                            Direction::Down => 'ო',
214                            Direction::Left => 'ɞ',
215                            Direction::Right => 'ʚ',
216                        }
217                    },
218                    Block::Placeable(PlaceableBlock::Air) => ' ',
219                };
220                let grapheme = grapheme::Id::from(char);
221
222                let bg_mutation = MutateBg(Set(bg_color));
223                let fg_mutation = MutateFg(Set(fg_color));
224                let color_mutation =
225                    MutateColors(bg_mutation.then(fg_mutation));
226                let grapheme_mutation = MutateGrapheme(Set(grapheme));
227                let mutation = color_mutation.then(grapheme_mutation);
228
229                app.canvas.queue([screen::Command::new_mutation(
230                    canvas_point,
231                    mutation,
232                )]);
233            }
234        }
235
236        Ok(())
237    }
238
239    fn border(&self) -> CoordPair {
240        self.feasible_min_freedom().zip2_with(
241            self.view.size,
242            |min_freedom, size| {
243                (size - min_freedom).min(self.config.border_max).max(1)
244            },
245        )
246    }
247
248    fn feasible_min_freedom(&self) -> CoordPair {
249        self.view
250            .size
251            .map(|coord| self.config.freedom_min.min(coord.saturating_sub(1)))
252    }
253
254    fn freedom_view(&self) -> Rect {
255        let border = self.border();
256        Rect {
257            top_left: self.view.top_left.saturating_add(&border),
258            size: self.view.size.saturating_sub(&(border * 2)),
259        }
260    }
261
262    fn center_on_player(&mut self, available_canvas: CoordPair, game: &Game) {
263        let view_size = available_canvas;
264        self.view = Rect {
265            top_left: game
266                .player()
267                .position()
268                .head()
269                .saturating_sub(&(view_size / 2)),
270            size: view_size,
271        };
272    }
273
274    fn stick_to_border(&mut self, game: &Game) {
275        let border = self.border();
276        let freedom_view = self.freedom_view();
277        let head = game.player().position().head();
278        let map_rect = game.map().rect();
279        self.view.top_left = CoordPair::from_axes(|axis| {
280            let start = if freedom_view.top_left[axis] > head[axis] {
281                head[axis].saturating_sub(border[axis])
282            } else if freedom_view.bottom_right()[axis] <= head[axis] {
283                head[axis]
284                    .saturating_sub(freedom_view.size[axis])
285                    .saturating_sub(border[axis])
286            } else {
287                self.view.top_left[axis]
288            };
289
290            start.max(map_rect.top_left[axis]).min(
291                map_rect.bottom_right()[axis]
292                    .saturating_sub(self.view.size[axis]),
293            )
294        });
295    }
296
297    fn available_canvas_size(
298        &mut self,
299        app: &mut App,
300        dynamic_style: &DynamicStyle,
301    ) -> Result<CoordPair, Error> {
302        let canvas_size = app.canvas.size();
303
304        let available = canvas_size
305            .checked_sub(&dynamic_style.margin_top_left)
306            .and_then(|size| {
307                size.checked_sub(&dynamic_style.margin_bottom_right)
308            })
309            .ok_or_else(|| InsufficientView {
310                canvas_size,
311                dynamic_style: dynamic_style.clone(),
312            })?;
313
314        Ok(available)
315    }
316}

376            margin_bottom_right: CoordPair { y: 0, x: 0 },

380            setup_resources(SetupArgs {
381                map_rect: Rect {

383                    size: CoordPair { x: 1000, y: 1050 },
384                },
385                player_head: CoordPair { y: 710, x: 1203 },
386                player_facing: Direction::Up,
387                camera: super::Config::default(),
388            });

390        device_mock.screen().enable_command_log();
391
392        let main = |mut app: App| async move {

394            app.canvas.flush().unwrap();

396            app.tick_session.tick().await;
