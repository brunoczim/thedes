106#[derive(Debug, Clone)]
107pub(crate) struct Camera {
108    view: Rect,
109    offset: CoordPair,
110    config: Config,
111}
112
113impl Camera {
114    fn new(config: Config) -> Self {
115        Self {
116            view: Rect {
117                top_left: CoordPair::from_axes(|_| 0),
118                size: CoordPair::from_axes(|_| 0),
119            },
120            offset: CoordPair { y: 1, x: 0 },
121            config,
122        }
123    }
124
125    pub fn update(&mut self, app: &mut App, game: &Game) {
126        if self.view.size != app.canvas.size() - self.offset {
127            self.center_on_player(app, game);
128        } else if !self
129            .freedom_view()
130            .contains_point(game.player().position().head())
131        {
132            self.stick_to_border(game);
133        } else if !self.view.contains_point(game.player().position().head())
134            || !self.view.contains_point(game.player().position().pointer())
135        {
136            self.center_on_player(app, game);
137        }
138    }
139
140    pub fn render(&mut self, app: &mut App, game: &Game) -> Result<(), Error> {
141        app.canvas
142            .queue([screen::Command::new_clear_screen(BasicColor::Black)]);
143
144        let pos_string = format!("↱{}", game.player().position().head());
145        text::styled(app, &pos_string, &text::Style::default())?;
146
147        for y in self.view.top_left.y .. self.view.bottom_right().y {
148            for x in self.view.top_left.x .. self.view.bottom_right().x {
149                let player_pos = game.player().position();
150                let point = CoordPair { y, x };
151                let canvas_point = point - self.view.top_left + self.offset;
152
153                let ground = game.map().get_ground(point)?;
154                let bg_color = match ground {
155                    Ground::Grass => Rgb::new(0x00, 0xff, 0x80).into(),
156                    Ground::Sand => Rgb::new(0xff, 0xff, 0x80).into(),
157                    Ground::Stone => Rgb::new(0xc0, 0xc0, 0xc0).into(),
158                };
159
160                let block = game.map().get_block(point)?;
161
162                let fg_color = BasicColor::Black.into();
163                let char = match block {
164                    Block::Special(SpecialBlock::Player) => {
165                        if player_pos.head() == point {
166                            'O'
167                        } else {
168                            match player_pos.facing() {
169                                Direction::Up => 'Ʌ',
170                                Direction::Down => 'V',
171                                Direction::Left => '<',
172                                Direction::Right => '>',
173                            }
174                        }
175                    },
176                    Block::Special(SpecialBlock::Monster(id)) => {
177                        let monster_pos =
178                            game.monster_registry().get_by_id(id)?.position();
179                        match monster_pos.facing() {
180                            Direction::Up => 'ɷ',
181                            Direction::Down => 'ო',
182                            Direction::Left => 'ɞ',
183                            Direction::Right => 'ʚ',
184                        }
185                    },
186                    Block::Placeable(PlaceableBlock::Air) => ' ',
187                };
188                let grapheme = grapheme::Id::from(char);
189
190                let bg_mutation = MutateBg(Set(bg_color));
191                let fg_mutation = MutateFg(Set(fg_color));
192                let color_mutation =
193                    MutateColors(bg_mutation.then(fg_mutation));
194                let grapheme_mutation = MutateGrapheme(Set(grapheme));
195                let mutation = color_mutation.then(grapheme_mutation);
196
197                app.canvas.queue([screen::Command::new_mutation(
198                    canvas_point,
199                    mutation,
200                )]);
201            }
202        }
203
204        Ok(())
205    }

207    fn border(&self) -> CoordPair {
208        self.feasible_min_freedom().zip2_with(
209            self.view.size,
210            |min_freedom, size| {
211                (size - min_freedom).min(self.config.border_max).max(1)
212            },
213        )
214    }
215
216    fn feasible_min_freedom(&self) -> CoordPair {
217        self.view
218            .size
219            .map(|coord| self.config.freedom_min.min(coord.saturating_sub(1)))
220    }
221
222    fn freedom_view(&self) -> Rect {
223        let border = self.border();
224        Rect {
225            top_left: self.view.top_left.saturating_add(&border),
226            size: self.view.size.saturating_sub(&(border * 2)),
227        }
228    }
229
230    fn center_on_player(&mut self, app: &mut App, game: &Game) {
231        let view_size = app.canvas.size() - self.offset;
232        self.view = Rect {
233            top_left: game
234                .player()
235                .position()
236                .head()
237                .saturating_sub(&(view_size / 2)),
238            size: view_size,
239        };
240    }
241
242    fn stick_to_border(&mut self, game: &Game) {
243        let border = self.border();
244        let freedom_view = self.freedom_view();
245        let head = game.player().position().head();
246        let map_rect = game.map().rect();
247        self.view.top_left = CoordPair::from_axes(|axis| {
248            let start = if freedom_view.top_left[axis] > head[axis] {
249                head[axis].saturating_sub(border[axis])
250            } else if freedom_view.bottom_right()[axis] <= head[axis] {
251                head[axis]
252                    .saturating_sub(freedom_view.size[axis])
253                    .saturating_sub(border[axis])
254            } else {
255                self.view.top_left[axis]
256            };
257
258            start.max(map_rect.top_left[axis]).min(
259                map_rect.bottom_right()[axis]
260                    .saturating_sub(self.view.size[axis]),
261            )
262        });
263    }

297        let game = Game::new(map, player_position).unwrap();

304                screen::Config::new()
305                    .with_canvas_size(CoordPair { y: 22, x: 78 }),
306            )
307            .with_device(device);
308
309        Resources { game, camera, runtime_config, device_mock }
310    }

312    #[tokio::test(flavor = "multi_thread")]
313    async fn correct_initial_view_min_commands() {
314        let Resources { game, mut camera, device_mock, runtime_config } =
315            setup_resources(SetupArgs {
316                map_rect: Rect {
317                    top_left: CoordPair { x: 500, y: 600 },
318                    size: CoordPair { x: 1000, y: 1050 },
319                },
320                player_head: CoordPair { y: 710, x: 1203 },
321                player_facing: Direction::Up,
322                camera: super::Config::default(),
323            });
324
325        device_mock.screen().enable_command_log();
326
327        let main = |mut app: App| async move {
328            camera.update(&mut app, &game);
329            camera.render(&mut app, &game).unwrap();
330            app.canvas.flush().unwrap();
331            app.tick_session.tick().await;
332            app.tick_session.tick().await;
333        };
334
335        let runtime_future = task::spawn(runtime_config.run(main));
336        tokio::time::sleep(Duration::from_millis(50)).await;
337        runtime_future.await.unwrap().unwrap();
338
339        let command_log =
340            device_mock.screen().take_command_log().unwrap().concat();
341
342        let expected_min_len = (22 - 1) * 78 + 1;
343        assert!(
344            command_log.len() >= expected_min_len,
345            "left: {}\nright: {}",
346            command_log.len(),
347            expected_min_len,
