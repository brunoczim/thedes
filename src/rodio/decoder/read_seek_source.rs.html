7/// A wrapper around a `Read + Seek` type that implements Symphonia's `MediaSource` trait.
8///
9/// This type allows standard Rust I/O types to be used with Symphonia's media framework
10/// by implementing the required `MediaSource` trait.
11pub struct ReadSeekSource<T: Read + Seek + Send + Sync> {
12    /// The wrapped reader/seeker
13    inner: T,
14    /// Optional length of the media source in bytes.
15    /// When known, this can help with seeking and duration calculations.
16    byte_len: Option<u64>,
17    /// Whether this media source reports as seekable.
18    is_seekable: bool,
19}
20
21impl<T: Read + Seek + Send + Sync> ReadSeekSource<T> {
22    /// Creates a new `ReadSeekSource` by wrapping a reader/seeker.
23    ///
24    /// # Arguments
25    /// * `inner` - The reader/seeker to wrap
26    /// * `settings` - Decoder settings for configuring the source
27    #[inline]
28    pub fn new(inner: T, settings: &Settings) -> Self {
29        ReadSeekSource {
30            inner,
31            byte_len: settings.byte_len,
32            is_seekable: settings.is_seekable,
33        }
34    }
35}
36
37impl<T: Read + Seek + Send + Sync> MediaSource for ReadSeekSource<T> {
38    /// Returns whether this media source reports as seekable.
39    #[inline]
40    fn is_seekable(&self) -> bool {
41        self.is_seekable
42    }
43
44    /// Returns the total length of the media source in bytes, if known.
45    #[inline]
46    fn byte_len(&self) -> Option<u64> {
47        self.byte_len
48    }
49}
50
51impl<T: Read + Seek + Send + Sync> Read for ReadSeekSource<T> {
52    #[inline]
53    /// Reads bytes from the underlying reader into the provided buffer.
54    ///
55    /// Delegates to the inner reader's implementation.
56    fn read(&mut self, buf: &mut [u8]) -> Result<usize> {
57        self.inner.read(buf)
58    }
59}
60
61impl<T: Read + Seek + Send + Sync> Seek for ReadSeekSource<T> {
62    /// Seeks to a position in the underlying reader.
63    ///
64    /// Delegates to the inner reader's implementation.
65    #[inline]
66    fn seek(&mut self, pos: SeekFrom) -> Result<u64> {
67        self.inner.seek(pos)
68    }
