13        #[from]
14        #[source]

20        #[source]

31#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
32pub enum Event {

35    TryMoveMonster(monster::Id, Direction),
36}

38impl Event {
39    pub fn apply(self, game: &mut Game) -> Result<(), ApplyError> {
40        match self {
41            Self::TrySpawnMonster(position) => {
42                game.try_spawn_moster(position)?
43            },
44            Self::VanishMonster(id) => game.vanish_monster(id)?,
45            Self::TryMoveMonster(id, direction) => {
46                game.try_move_monster(id, direction)?
47            },
48        }
49        Ok(())
50    }
