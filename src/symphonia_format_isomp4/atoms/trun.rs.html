14/// Track fragment run atom.
15#[allow(dead_code)]
16#[derive(Debug)]
17pub struct TrunAtom {
18    /// Atom header.
19    header: AtomHeader,
20    /// Extended header flags.
21    flags: u32,
22    /// Data offset of this run.
23    pub data_offset: Option<i32>,
24    /// Number of samples in this run.
25    pub sample_count: u32,
26    /// Sample flags for the first sample only.
27    pub first_sample_flags: Option<u32>,
28    /// Sample duration for each sample in this run.
29    pub sample_duration: Vec<u32>,
30    /// Sample size for each sample in this run.
31    pub sample_size: Vec<u32>,
32    /// Sample flags for each sample in this run.
33    pub sample_flags: Vec<u32>,
34    /// The total size of all samples in this run. 0 if the sample size flag is not set.
35    total_sample_size: u64,
36    /// The total duration of all samples in this run. 0 if the sample duration flag is not set.
37    total_sample_duration: u64,
38}
39
40impl TrunAtom {
41    // Track fragment run atom flags.
42    const DATA_OFFSET_PRESENT: u32 = 0x1;
43    const FIRST_SAMPLE_FLAGS_PRESENT: u32 = 0x4;
44    const SAMPLE_DURATION_PRESENT: u32 = 0x100;
45    const SAMPLE_SIZE_PRESENT: u32 = 0x200;
46    const SAMPLE_FLAGS_PRESENT: u32 = 0x400;
47    const SAMPLE_COMPOSITION_TIME_OFFSETS_PRESENT: u32 = 0x800;
48
49    /// Indicates if sample durations are provided.
50    pub fn is_sample_duration_present(&self) -> bool {
51        self.flags & TrunAtom::SAMPLE_DURATION_PRESENT != 0
52    }
53
54    // Indicates if the duration of the first sample is provided.
55    pub fn is_first_sample_duration_present(&self) -> bool {
56        match self.first_sample_flags {
57            Some(flags) => flags & TrunAtom::FIRST_SAMPLE_FLAGS_PRESENT != 0,
58            None => false,
59        }
60    }
61
62    /// Indicates if sample sizes are provided.
63    pub fn is_sample_size_present(&self) -> bool {
64        self.flags & TrunAtom::SAMPLE_SIZE_PRESENT != 0
65    }
66
67    /// Indicates if the size for the first sample is provided.
68    pub fn is_first_sample_size_present(&self) -> bool {
69        match self.first_sample_flags {
70            Some(flags) => flags & TrunAtom::SAMPLE_SIZE_PRESENT != 0,
71            None => false,
72        }
73    }
74
75    /// Indicates if sample flags are provided.
76    #[allow(dead_code)]
77    pub fn are_sample_flags_present(&self) -> bool {
78        self.flags & TrunAtom::SAMPLE_FLAGS_PRESENT != 0
79    }
80
81    /// Indicates if sample composition time offsets are provided.
82    #[allow(dead_code)]
83    pub fn are_sample_composition_time_offsets_present(&self) -> bool {
84        self.flags & TrunAtom::SAMPLE_COMPOSITION_TIME_OFFSETS_PRESENT != 0
85    }
86
87    /// Gets the total duration of all samples.
88    pub fn total_duration(&self, default_dur: u32) -> u64 {
89        if self.is_sample_duration_present() {
90            self.total_sample_duration
91        }
92        else {
93            // The duration of all samples in the track fragment are not explictly known.
94            if self.sample_count > 0 && self.is_first_sample_duration_present() {
95                // The first sample has an explictly recorded duration.
96                u64::from(self.sample_duration[0])
97                    + u64::from(self.sample_count - 1) * u64::from(default_dur)
98            }
99            else {
100                // All samples have the default duration.
101                u64::from(self.sample_count) * u64::from(default_dur)
102            }
103        }
104    }
105
106    /// Gets the total size of all samples.
107    pub fn total_size(&self, default_size: u32) -> u64 {
108        if self.is_sample_size_present() {
109            self.total_sample_size
110        }
111        else if self.sample_count > 0 && self.is_first_sample_size_present() {
112            u64::from(self.sample_size[0])
113                + u64::from(self.sample_count - 1) * u64::from(default_size)
114        }
115        else {
116            u64::from(self.sample_count) * u64::from(default_size)
117        }
118    }
119
120    /// Get the timestamp and duration of a sample. The desired sample is specified by the
121    /// trun-relative sample number, `sample_num_rel`.
122    pub fn sample_timing(&self, sample_num_rel: u32, default_dur: u32) -> (u64, u32) {
123        debug_assert!(sample_num_rel < self.sample_count);
124
125        if self.is_sample_duration_present() {
126            // All sample durations are unique.
127            let ts = if sample_num_rel > 0 {
128                self.sample_duration[..sample_num_rel as usize]
129                    .iter()
130                    .map(|&s| u64::from(s))
131                    .sum::<u64>()
132            }
133            else {
134                0
135            };
136
137            let dur = self.sample_duration[sample_num_rel as usize];
138
139            (ts, dur)
140        }
141        else {
142            // The duration of all samples in the track fragment are not unique.
143            let ts = if sample_num_rel > 0 && self.is_first_sample_duration_present() {
144                // The first sample has a unique duration.
145                u64::from(self.sample_duration[0])
146                    + u64::from(sample_num_rel - 1) * u64::from(default_dur)
147            }
148            else {
149                // Zero or more samples with identical durations.
150                u64::from(sample_num_rel) * u64::from(default_dur)
151            };
152
153            (ts, default_dur)
154        }
155    }
156
157    /// Get the size of a sample. The desired sample is specified by the trun-relative sample
158    /// number, `sample_num_rel`.
159    pub fn sample_size(&self, sample_num_rel: u32, default_size: u32) -> u32 {
160        debug_assert!(sample_num_rel < self.sample_count);
161
162        if self.is_sample_size_present() {
163            self.sample_size[sample_num_rel as usize]
164        }
165        else if sample_num_rel == 0 && self.is_first_sample_size_present() {
166            self.sample_size[0]
167        }
168        else {
169            default_size
170        }
171    }
172
173    /// Get the byte offset and size of a sample. The desired sample is specified by the
174    /// trun-relative sample number, `sample_num_rel`.
175    pub fn sample_offset(&self, sample_num_rel: u32, default_size: u32) -> (u64, u32) {
176        debug_assert!(sample_num_rel < self.sample_count);
177
178        if self.is_sample_size_present() {
179            // All sample sizes are unique.
180            let offset = if sample_num_rel > 0 {
181                self.sample_size[..sample_num_rel as usize]
182                    .iter()
183                    .map(|&s| u64::from(s))
184                    .sum::<u64>()
185            }
186            else {
187                0
188            };
189
190            (offset, self.sample_size[sample_num_rel as usize])
191        }
192        else {
193            // The size of all samples in the track are not unique.
194            let offset = if sample_num_rel > 0 && self.is_first_sample_size_present() {
195                // The first sample has a unique size.
196                u64::from(self.sample_size[0])
197                    + u64::from(sample_num_rel - 1) * u64::from(default_size)
198            }
199            else {
200                // Zero or more identically sized samples.
201                u64::from(sample_num_rel) * u64::from(default_size)
202            };
203
204            (offset, default_size)
205        }
206    }
207
208    /// Get the sample number (relative to the trun) of the sample that contains timestamp `ts`.
209    pub fn ts_sample(&self, ts_rel: u64, default_dur: u32) -> u32 {
210        let mut sample_num = 0;
211        let mut ts_delta = ts_rel;
212
213        if self.is_sample_duration_present() {
214            // If the sample durations are present, then each sample duration is independently
215            // stored. Sum sample durations until the delta is reached.
216            for &dur in &self.sample_duration {
217                if u64::from(dur) > ts_delta {
218                    break;
219                }
220
221                ts_delta -= u64::from(dur);
222                sample_num += 1;
223            }
224        }
225        else {
226            if self.sample_count > 0 && self.is_first_sample_duration_present() {
227                // The first sample duration is unique.
228                let first_sample_dur = u64::from(self.sample_duration[0]);
229
230                if ts_delta >= first_sample_dur {
231                    ts_delta -= first_sample_dur;
232                    sample_num += 1;
233                }
234                else {
235                    ts_delta -= ts_delta;
236                }
237            }
238
239            sample_num += ts_delta.checked_div(u64::from(default_dur)).unwrap_or(0) as u32;
240        }
241
242        sample_num
243    }
244}
245
246impl Atom for TrunAtom {
247    fn header(&self) -> AtomHeader {
248        self.header
249    }
250
251    fn read<B: ReadBytes>(reader: &mut B, header: AtomHeader) -> Result<Self> {
252        let (_, flags) = AtomHeader::read_extra(reader)?;
253
254        let sample_count = reader.read_be_u32()?;
255
256        let data_offset = match flags & TrunAtom::DATA_OFFSET_PRESENT {
257            0 => None,
258            _ => Some(bits::sign_extend_leq32_to_i32(reader.read_be_u32()?, 32)),
259        };
260
261        let first_sample_flags = match flags & TrunAtom::FIRST_SAMPLE_FLAGS_PRESENT {
262            0 => None,
263            _ => Some(reader.read_be_u32()?),
264        };
265
266        // If the first-sample-flags-present flag is set, then the sample-flags-present flag should
267        // not be set. The samples after the first shall use the default sample flags defined in the
268        // tfhd or mvex atoms.
269        if first_sample_flags.is_some() && (flags & TrunAtom::SAMPLE_FLAGS_PRESENT != 0) {
270            return decode_error(
271                "isomp4: sample-flag-present and first-sample-flags-present flags are set",
272            );
273        }
274
275        let mut sample_duration = Vec::new();
276        let mut sample_size = Vec::new();
277        let mut sample_flags = Vec::new();
278
279        let mut total_sample_size = 0;
280        let mut total_sample_duration = 0;
281
282        // TODO: Apply a limit.
283        for _ in 0..sample_count {
284            if (flags & TrunAtom::SAMPLE_DURATION_PRESENT) != 0 {
285                let duration = reader.read_be_u32()?;
286                total_sample_duration += u64::from(duration);
287                sample_duration.push(duration);
288            }
289
290            if (flags & TrunAtom::SAMPLE_SIZE_PRESENT) != 0 {
291                let size = reader.read_be_u32()?;
292                total_sample_size += u64::from(size);
293                sample_size.push(size);
294            }
295
296            if (flags & TrunAtom::SAMPLE_FLAGS_PRESENT) != 0 {
297                sample_flags.push(reader.read_be_u32()?);
298            }
299
300            // Ignoring composition time for now since it's a video thing...
301            if (flags & TrunAtom::SAMPLE_COMPOSITION_TIME_OFFSETS_PRESENT) != 0 {
302                // For version 0, this is a u32.
303                // For version 1, this is a i32.
304                let _ = reader.read_be_u32()?;
305            }
306        }
307
308        Ok(TrunAtom {
309            header,
310            flags,
311            data_offset,
312            sample_count,
313            first_sample_flags,
314            sample_duration,
315            sample_size,
316            sample_flags,
317            total_sample_size,
318            total_sample_duration,
319        })
320    }
