8use symphonia_core::codecs::{CodecParameters, VerificationCheck, CODEC_TYPE_FLAC};
9use symphonia_core::errors::{decode_error, unsupported_error, Result};
10use symphonia_core::io::{BufReader, ReadBytes};
11
12use symphonia_utils_xiph::flac::metadata::{MetadataBlockHeader, MetadataBlockType, StreamInfo};
13
14use crate::atoms::{Atom, AtomHeader};
15
16#[allow(dead_code)]
17#[derive(Debug)]
18pub struct FlacAtom {
19    /// Atom header.
20    header: AtomHeader,
21    /// FLAC stream info block.
22    stream_info: StreamInfo,
23    /// FLAC extra data.
24    extra_data: Box<[u8]>,
25}
26
27impl Atom for FlacAtom {
28    fn header(&self) -> AtomHeader {
29        self.header
30    }
31
32    fn read<B: ReadBytes>(reader: &mut B, header: AtomHeader) -> Result<Self> {
33        let (version, flags) = AtomHeader::read_extra(reader)?;
34
35        if version != 0 {
36            return unsupported_error("isomp4 (flac): unsupported flac version");
37        }
38
39        if flags != 0 {
40            return decode_error("isomp4 (flac): flags not zero");
41        }
42
43        // The first block must be the stream information block.
44        let block_header = MetadataBlockHeader::read(reader)?;
45
46        if block_header.block_type != MetadataBlockType::StreamInfo {
47            return decode_error("isomp4 (flac): first block is not stream info");
48        }
49
50        // Ensure the block length is correct for a stream information block before allocating a
51        // buffer for it.
52        if !StreamInfo::is_valid_size(u64::from(block_header.block_len)) {
53            return decode_error("isomp4 (flac): invalid stream info block length");
54        }
55
56        let extra_data = reader.read_boxed_slice_exact(block_header.block_len as usize)?;
57        let stream_info = StreamInfo::read(&mut BufReader::new(&extra_data))?;
58
59        Ok(FlacAtom { header, stream_info, extra_data })
60    }
61}
62
63impl FlacAtom {
64    pub fn fill_codec_params(&self, codec_params: &mut CodecParameters) {
65        codec_params
66            .for_codec(CODEC_TYPE_FLAC)
67            .with_sample_rate(self.stream_info.sample_rate)
68            .with_bits_per_sample(self.stream_info.bits_per_sample)
69            .with_channels(self.stream_info.channels)
70            .with_packet_data_integrity(true)
71            .with_extra_data(self.extra_data.clone());
72
73        if let Some(md5) = self.stream_info.md5 {
74            codec_params.with_verification_code(VerificationCheck::Md5(md5));
75        }
76    }
77}
