29pub struct TrackState {
30    codec_params: CodecParameters,
31    /// The track number.
32    track_num: usize,
33    /// The current segment.
34    cur_seg: usize,
35    /// The current sample index relative to the track.
36    next_sample: u32,
37    /// The current sample byte position relative to the start of the track.
38    next_sample_pos: u64,
39}

41impl TrackState {
42    #[allow(clippy::single_match)]
43    pub fn new(track_num: usize, trak: &TrakAtom) -> Self {
44        let mut codec_params = CodecParameters::new();
45
46        codec_params
47            .with_time_base(TimeBase::new(1, trak.mdia.mdhd.timescale))
48            .with_n_frames(trak.mdia.mdhd.duration);
49
50        // Fill the codec parameters using the sample description atom.
51        trak.mdia.minf.stbl.stsd.fill_codec_params(&mut codec_params);
52
53        Self { codec_params, track_num, cur_seg: 0, next_sample: 0, next_sample_pos: 0 }
54    }
55
56    pub fn codec_params(&self) -> CodecParameters {
57        self.codec_params.clone()
58    }

61/// Information regarding the next sample.
62#[derive(Debug)]
63struct NextSampleInfo {
64    /// The track number of the next sample.
65    track_num: usize,
66    /// The timestamp of the next sample.
67    ts: u64,
68    /// The timestamp expressed in seconds.
69    time: Time,
70    /// The duration of the next sample.
71    dur: u32,
72    /// The segment containing the next sample.
73    seg_idx: usize,
74}
75
76/// Information regarding a sample.
77#[derive(Debug)]
78struct SampleDataInfo {
79    /// The position of the sample in the track.
80    pos: u64,
81    /// The length of the sample.
82    len: u32,
83}

85/// ISO Base Media File Format (MP4, M4A, MOV, etc.) demultiplexer.
86///
87/// `IsoMp4Reader` implements a demuxer for the ISO Base Media File Format.
88pub struct IsoMp4Reader {
89    iter: AtomIterator<MediaSourceStream>,
90    tracks: Vec<Track>,
91    cues: Vec<Cue>,
92    metadata: MetadataLog,
93    /// Segments of the movie. Sorted in ascending order by sequence number.
94    segs: Vec<Box<dyn StreamSegment>>,
95    /// State tracker for each track.
96    track_states: Vec<TrackState>,
97    /// Optional, movie extends atom used for fragmented streams.
98    mvex: Option<Arc<MvexAtom>>,
99}

101impl IsoMp4Reader {
102    /// Idempotently gets information regarding the next sample of the media stream. This function
103    /// selects the next sample with the lowest timestamp of all tracks.
104    fn next_sample_info(&self) -> Result<Option<NextSampleInfo>> {
105        let mut earliest = None;
106
107        // TODO: Consider returning samples based on lowest byte position in the track instead of
108        // timestamp. This may be important if video tracks are ever decoded (i.e., DTS vs. PTS).
109
110        for (state, track) in self.track_states.iter().zip(&self.tracks) {
111            // Get the timebase of the track used to calculate the presentation time.
112            let tb = track.codec_params.time_base.unwrap();
113
114            // Get the next timestamp for the next sample of the current track. The next sample may
115            // be in a future segment.
116            for (seg_idx_delta, seg) in self.segs[state.cur_seg..].iter().enumerate() {
117                // Try to get the timestamp for the next sample of the track from the segment.
118                if let Some(timing) = seg.sample_timing(state.track_num, state.next_sample)? {
119                    // Calculate the presentation time using the timestamp.
120                    let sample_time = tb.calc_time(timing.ts);
121
122                    // Compare the presentation time of the sample from this track to other tracks,
123                    // and select the track with the earliest presentation time.
124                    match earliest {
125                        Some(NextSampleInfo { track_num: _, ts: _, time, dur: _, seg_idx: _ })
126                            if time <= sample_time =>
127                        {
128                            // Earliest is less than or equal to the track's next sample
129                            // presentation time. No need to update earliest.
130                        }
131                        _ => {
132                            // Earliest was either None, or greater than the track's next sample
133                            // presentation time. Update earliest.
134                            earliest = Some(NextSampleInfo {
135                                track_num: state.track_num,
136                                ts: timing.ts,
137                                time: sample_time,
138                                dur: timing.dur,
139                                seg_idx: seg_idx_delta + state.cur_seg,
140                            });
141                        }
142                    }
143
144                    // Either the next sample of the track had the earliest presentation time seen
145                    // thus far, or it was greater than those from other tracks, but there is no
146                    // reason to check samples in future segments.
147                    break;
148                }
149            }
150        }
151
152        Ok(earliest)
153    }
154
155    fn consume_next_sample(&mut self, info: &NextSampleInfo) -> Result<Option<SampleDataInfo>> {
156        // Get the track state.
157        let track = &mut self.track_states[info.track_num];
158
159        // Get the segment associated with the sample.
160        let seg = &self.segs[info.seg_idx];
161
162        // Get the sample data descriptor.
163        let sample_data_desc = seg.sample_data(track.track_num, track.next_sample, false)?;
164
165        // The sample base position in the sample data descriptor remains constant if the sample
166        // followed immediately after the previous sample. In this case, the track state's
167        // next_sample_pos is the position of the current sample. If the base position has jumped,
168        // then the base position is the position of current the sample.
169        let pos = if sample_data_desc.base_pos > track.next_sample_pos {
170            sample_data_desc.base_pos
171        }
172        else {
173            track.next_sample_pos
174        };
175
176        // Advance the track's current segment to the next sample's segment.
177        track.cur_seg = info.seg_idx;
178
179        // Advance the track's next sample number and position.
180        track.next_sample += 1;
181        track.next_sample_pos = pos + u64::from(sample_data_desc.size);
182
183        Ok(Some(SampleDataInfo { pos, len: sample_data_desc.size }))
184    }
185
186    fn try_read_more_segments(&mut self) -> Result<()> {
187        // Continue iterating over atoms until a segment (a moof + mdat atom pair) is found. All
188        // other atoms will be ignored.
189        while let Some(header) = self.iter.next_no_consume()? {
190            match header.atype {
191                AtomType::MediaData => {
192                    // Consume the atom from the iterator so that on the next iteration a new atom
193                    // will be read.
194                    self.iter.consume_atom();
195
196                    return Ok(());
197                }
198                AtomType::MovieFragment => {
199                    let moof = self.iter.read_atom::<MoofAtom>()?;
200
201                    // A moof segment can only be created if the mvex atom is present.
202                    if let Some(mvex) = &self.mvex {
203                        // Get the last segment. Note, there will always be one segment because the
204                        // moov atom is converted into a segment when the reader is instantiated.
205                        let last_seg = self.segs.last().unwrap();
206
207                        // Create a new segment for the moof atom.
208                        let seg = MoofSegment::new(moof, mvex.clone(), last_seg.as_ref());
209
210                        // Segments should have a monotonic sequence number.
211                        if seg.sequence_num() <= last_seg.sequence_num() {
212                            warn!("moof fragment has a non-monotonic sequence number.");
213                        }
214
215                        // Push the segment.
216                        self.segs.push(Box::new(seg));
217                    }
218                    else {
219                        // TODO: This is a fatal error.
220                        return decode_error("isomp4: moof atom present without mvex atom");
221                    }
222                }
223                _ => {
224                    trace!("skipping atom: {:?}.", header.atype);
225                    self.iter.consume_atom();
226                }
227            }
228        }
229
230        // If no atoms were returned above, then the end-of-stream has been reached.
231        end_of_stream_error()
232    }

234    fn seek_track_by_time(&mut self, track_num: usize, time: Time) -> Result<SeekedTo> {
235        // Convert time to timestamp for the track.
236        if let Some(track) = self.tracks.get(track_num) {
237            let tb = track.codec_params.time_base.unwrap();
238            self.seek_track_by_ts(track_num, tb.calc_timestamp(time))
239        }
240        else {
241            seek_error(SeekErrorKind::Unseekable)
242        }
243    }
244
245    fn seek_track_by_ts(&mut self, track_num: usize, ts: u64) -> Result<SeekedTo> {
246        debug!("seeking track={} to frame_ts={}", track_num, ts);
247
248        struct SeekLocation {
249            seg_idx: usize,
250            sample_num: u32,
251        }
252
253        let mut seek_loc = None;
254        let mut seg_skip = 0;
255
256        loop {
257            // Iterate over all segments and attempt to find the segment and sample number that
258            // contains the desired timestamp. Skip segments already examined.
259            for (seg_idx, seg) in self.segs.iter().enumerate().skip(seg_skip) {
260                if let Some(sample_num) = seg.ts_sample(track_num, ts)? {
261                    seek_loc = Some(SeekLocation { seg_idx, sample_num });
262                    break;
263                }
264
265                // Mark the segment as examined.
266                seg_skip = seg_idx + 1;
267            }
268
269            // If a seek location is found, break.
270            if seek_loc.is_some() {
271                break;
272            }
273
274            // Otherwise, try to read more segments from the stream.
275            self.try_read_more_segments()?;
276        }
277
278        if let Some(seek_loc) = seek_loc {
279            let seg = &self.segs[seek_loc.seg_idx];
280
281            // Get the sample information.
282            let data_desc = seg.sample_data(track_num, seek_loc.sample_num, true)?;
283
284            // Update the track's next sample information to point to the seeked sample.
285            let track = &mut self.track_states[track_num];
286
287            track.cur_seg = seek_loc.seg_idx;
288            track.next_sample = seek_loc.sample_num;
289            track.next_sample_pos = data_desc.base_pos + data_desc.offset.unwrap();
290
291            // Get the actual timestamp for this sample.
292            let timing = seg.sample_timing(track_num, seek_loc.sample_num)?.unwrap();
293
294            debug!(
295                "seeked track={} to packet_ts={} (delta={})",
296                track_num,
297                timing.ts,
298                timing.ts as i64 - ts as i64
299            );
300
301            Ok(SeekedTo { track_id: track_num as u32, required_ts: ts, actual_ts: timing.ts })
302        }
303        else {
304            // Timestamp was not found.
305            seek_error(SeekErrorKind::OutOfRange)
306        }
307    }

326impl FormatReader for IsoMp4Reader {
327    fn try_new(mut mss: MediaSourceStream, _options: &FormatOptions) -> Result<Self> {
328        // To get to beginning of the atom.
329        mss.seek_buffered_rel(-4);
330
331        let is_seekable = mss.is_seekable();
332
333        let mut ftyp = None;
334        let mut moov = None;
335        let mut sidx = None;
336
337        // Get the total length of the stream, if possible.
338        let total_len = if is_seekable {
339            let pos = mss.pos();
340            let len = mss.byte_len().ok_or(Error::SeekError(SeekErrorKind::Unseekable))?;
341            mss.seek(SeekFrom::Start(pos))?;
342            info!("stream is seekable with len={} bytes.", len);
343            Some(len)
344        }
345        else {
346            None
347        };
348
349        let mut metadata = MetadataLog::default();
350
351        // Parse all atoms if the stream is seekable, otherwise parse all atoms up-to the mdat atom.
352        let mut iter = AtomIterator::new_root(mss, total_len);
353
354        while let Some(header) = iter.next()? {
355            // Top-level atoms.
356            match header.atype {
357                AtomType::FileType => {
358                    ftyp = Some(iter.read_atom::<FtypAtom>()?);
359                }
360                AtomType::Movie => {
361                    moov = Some(iter.read_atom::<MoovAtom>()?);
362                }
363                AtomType::SegmentIndex => {
364                    // If the stream is not seekable, then it can only be assumed that the first
365                    // segment index atom is indeed the first segment index because the format
366                    // reader cannot practically skip past this point.
367                    if !is_seekable {
368                        sidx = Some(iter.read_atom::<SidxAtom>()?);
369                        break;
370                    }
371                    else {
372                        // If the stream is seekable, examine all segment indexes and select the
373                        // index with the earliest presentation timestamp to be the first.
374                        let new_sidx = iter.read_atom::<SidxAtom>()?;
375
376                        let is_earlier = match &sidx {
377                            Some(sidx) => new_sidx.earliest_pts < sidx.earliest_pts,
378                            _ => true,
379                        };
380
381                        if is_earlier {
382                            sidx = Some(new_sidx);
383                        }
384                    }
385                }
386                AtomType::MediaData | AtomType::MovieFragment => {
387                    // The mdat atom contains the codec bitstream data. For segmented streams, a
388                    // moof + mdat pair is required for playback. If the source is unseekable then
389                    // the format reader cannot skip past these atoms without dropping samples.
390                    if !is_seekable {
391                        // If the moov atom hasn't been seen before the moof and/or mdat atom, and
392                        // the stream is not seekable, then the mp4 is not streamable.
393                        if moov.is_none() || ftyp.is_none() {
394                            warn!("mp4 is not streamable.");
395                        }
396
397                        // The remainder of the stream will be read incrementally.
398                        break;
399                    }
400                }
401                AtomType::Meta => {
402                    // Read the metadata atom and append it to the log.
403                    let mut meta = iter.read_atom::<MetaAtom>()?;
404
405                    if let Some(rev) = meta.take_metadata() {
406                        metadata.push(rev);
407                    }
408                }
409                AtomType::Free => (),
410                AtomType::Skip => (),
411                _ => {
412                    info!("skipping top-level atom: {:?}.", header.atype);
413                }
414            }
415        }
416
417        if ftyp.is_none() {
418            return unsupported_error("isomp4: missing ftyp atom");
419        }
420
421        if moov.is_none() {
422            return unsupported_error("isomp4: missing moov atom");
423        }
424
425        // If the stream was seekable, then all atoms in the media source stream were scanned. Seek
426        // back to the first mdat atom for playback. If the stream is not seekable, then the atom
427        // iterator is currently positioned at the first mdat atom.
428        if is_seekable {
429            let mut mss = iter.into_inner();
430            mss.seek(SeekFrom::Start(0))?;
431
432            iter = AtomIterator::new_root(mss, total_len);
433
434            while let Some(header) = iter.next_no_consume()? {
435                match header.atype {
436                    AtomType::MediaData | AtomType::MovieFragment => break,
437                    _ => (),
438                }
439                iter.consume_atom();
440            }
441        }
442
443        let mut moov = moov.unwrap();
444
445        if moov.is_fragmented() {
446            // If a Segment Index (sidx) atom was found, add the segments contained within.
447            if sidx.is_some() {
448                info!("stream is segmented with a segment index.");
449            }
450            else {
451                info!("stream is segmented without a segment index.");
452            }
453        }
454
455        if let Some(rev) = moov.take_metadata() {
456            metadata.push(rev);
457        }
458
459        // Instantiate a TrackState for each track in the stream.
460        let track_states = moov
461            .traks
462            .iter()
463            .enumerate()
464            .map(|(t, trak)| TrackState::new(t, trak))
465            .collect::<Vec<TrackState>>();
466
467        // Instantiate a Tracks for all tracks above.
468        let tracks = track_states
469            .iter()
470            .map(|track| Track::new(track.track_num as u32, track.codec_params()))
471            .collect();
472
473        // A Movie Extends (mvex) atom is required to support segmented streams. If the mvex atom is
474        // present, wrap it in an Arc so it can be shared amongst all segments.
475        let mvex = moov.mvex.take().map(Arc::new);
476
477        // The number of tracks specified in the moov atom must match the number in the mvex atom.
478        if let Some(mvex) = &mvex {
479            if mvex.trexs.len() != moov.traks.len() {
480                return decode_error("isomp4: mvex and moov track number mismatch");
481            }
482        }
483
484        let segs: Vec<Box<dyn StreamSegment>> = vec![Box::new(MoovSegment::new(moov))];
485
486        Ok(IsoMp4Reader {
487            iter,
488            tracks,
489            cues: Default::default(),
490            metadata,
491            track_states,
492            segs,
493            mvex,
494        })
495    }
496
497    fn next_packet(&mut self) -> Result<Packet> {
498        // Get the index of the track with the next-nearest (minimum) timestamp.
499        let next_sample_info = loop {
500            // Using the current set of segments, try to get the next sample info.
501            if let Some(info) = self.next_sample_info()? {
502                break info;
503            }
504            else {
505                // No more segments. If the stream is unseekable, it may be the case that there are
506                // more segments coming. Iterate atoms until a new segment is found or the
507                // end-of-stream is reached.
508                self.try_read_more_segments()?;
509            }
510        };
511
512        // Get the position and length information of the next sample.
513        let sample_info = self.consume_next_sample(&next_sample_info)?.unwrap();
514
515        let reader = self.iter.inner_mut();
516
517        // Attempt a fast seek within the buffer cache.
518        if reader.seek_buffered(sample_info.pos) != sample_info.pos {
519            if reader.is_seekable() {
520                // Fallback to a slow seek if the stream is seekable.
521                reader.seek(SeekFrom::Start(sample_info.pos))?;
522            }
523            else if sample_info.pos > reader.pos() {
524                // The stream is not seekable but the desired seek position is ahead of the reader's
525                // current position, thus the seek can be emulated by ignoring the bytes up to the
526                // the desired seek position.
527                reader.ignore_bytes(sample_info.pos - reader.pos())?;
528            }
529            else {
530                // The stream is not seekable and the desired seek position falls outside the lower
531                // bound of the buffer cache. This sample cannot be read.
532                return decode_error("isomp4: packet out-of-bounds for a non-seekable stream");
533            }
534        }
535
536        Ok(Packet::new_from_boxed_slice(
537            next_sample_info.track_num as u32,
538            next_sample_info.ts,
539            u64::from(next_sample_info.dur),
540            reader.read_boxed_slice_exact(sample_info.len as usize)?,
541        ))
542    }

556    fn seek(&mut self, _mode: SeekMode, to: SeekTo) -> Result<SeekedTo> {
557        if self.tracks.is_empty() {
558            return seek_error(SeekErrorKind::Unseekable);
559        }
560
561        match to {
562            SeekTo::TimeStamp { ts, track_id } => {
563                let selected_track_id = track_id as usize;
564
565                // The seek timestamp is in timebase units specific to the selected track. Get the
566                // selected track and use the timebase to convert the timestamp into time units so
567                // that the other tracks can be seeked.
568                if let Some(selected_track) = self.tracks().get(selected_track_id) {
569                    // Convert to time units.
570                    let time = selected_track.codec_params.time_base.unwrap().calc_time(ts);
571
572                    // Seek all tracks excluding the primary track to the desired time.
573                    for t in 0..self.track_states.len() {
574                        if t != selected_track_id {
575                            self.seek_track_by_time(t, time)?;
576                        }
577                    }
578
579                    // Seek the primary track and return the result.
580                    self.seek_track_by_ts(selected_track_id, ts)
581                }
582                else {
583                    seek_error(SeekErrorKind::Unseekable)
584                }
585            }
586            SeekTo::Time { time, track_id } => {
587                // Select the first track if a selected track was not provided.
588                let selected_track_id = track_id.unwrap_or(0) as usize;
589
590                // Seek all tracks excluding the selected track and discard the result.
591                for t in 0..self.track_states.len() {
592                    if t != selected_track_id {
593                        self.seek_track_by_time(t, time)?;
594                    }
595                }
596
597                // Seek the primary track and return the result.
598                self.seek_track_by_time(selected_track_id, time)
599            }
600        }
601    }

604        self.iter.into_inner()
605    }
