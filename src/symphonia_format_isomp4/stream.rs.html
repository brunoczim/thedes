14/// Sample data information.
15pub struct SampleDataDesc {
16    /// The starting byte position within the media data of the group of samples that contains the
17    /// sample described.
18    pub base_pos: u64,
19    /// The offset relative to the base position of the sample described.
20    pub offset: Option<u64>,
21    /// The size of the sample.
22    pub size: u32,
23}
24
25/// Timing information for one sample.
26pub struct SampleTiming {
27    /// The timestamp of the sample.
28    pub ts: u64,
29    /// The duration of the sample.
30    pub dur: u32,
31}
32
33pub trait StreamSegment: Send + Sync {
34    /// Gets the sequence number of this segment.
35    fn sequence_num(&self) -> u32;
36
37    /// Gets the first and last sample numbers for the track `track_num`.
38    fn track_sample_range(&self, track_num: usize) -> Range<u32>;
39
40    /// Gets the first and last sample timestamps for the track `track_num`.
41    fn track_ts_range(&self, track_num: usize) -> Range<u64>;
42
43    /// Get the timestamp and duration for the sample indicated by `sample_num` for the track
44    /// `track_num`.
45    fn sample_timing(&self, track_num: usize, sample_num: u32) -> Result<Option<SampleTiming>>;
46
47    /// Get the sample number of the sample containing the timestamp indicated by `ts` for track
48    // `track_num`.
49    fn ts_sample(&self, track_num: usize, ts: u64) -> Result<Option<u32>>;
50
51    /// Get the byte position of the group of samples containing the sample indicated by
52    /// `sample_num` for track `track_num`, and it's size.
53    ///
54    /// Optionally, the offset of the sample relative to the aforementioned byte position can be
55    /// returned.
56    fn sample_data(
57        &self,
58        track_num: usize,
59        sample_num: u32,
60        get_offset: bool,
61    ) -> Result<SampleDataDesc>;
62}

64/// Track-to-stream sequencing information.
65#[derive(Copy, Clone, Debug, Default)]
66struct SequenceInfo {
67    /// The sample number of the first sample of a track in a fragment.
68    first_sample: u32,
69    /// The timestamp of the first sample of a track in a fragment.
70    first_ts: u64,
71    /// The total duration of all samples of a track in a fragment.
72    total_sample_duration: u64,
73    /// The total sample count of a track in a fragment.
74    total_sample_count: u32,
75    /// If present in the moof segment, this is the index of the track fragment atom for the track
76    /// this sequence information is associated with.
77    traf_idx: Option<usize>,
78}
79
80pub struct MoofSegment {
81    moof: MoofAtom,
82    mvex: Arc<MvexAtom>,
83    seq: Vec<SequenceInfo>,
84}
85
86impl MoofSegment {
87    /// Instantiate a new segment from a `MoofAtom`.
88    pub fn new(moof: MoofAtom, mvex: Arc<MvexAtom>, prev: &dyn StreamSegment) -> MoofSegment {
89        let mut seq = Vec::with_capacity(mvex.trexs.len());
90
91        // Calculate the sequence information for each track, even if not present in the fragment.
92        for (track_num, trex) in mvex.trexs.iter().enumerate() {
93            let mut info = SequenceInfo {
94                first_sample: prev.track_sample_range(track_num).end,
95                first_ts: prev.track_ts_range(track_num).end,
96                ..Default::default()
97            };
98
99            // Find the track fragment for the track.
100            for (traf_idx, traf) in moof.trafs.iter().enumerate() {
101                if trex.track_id != traf.tfhd.track_id {
102                    continue;
103                }
104
105                // Calculate the total duration of all runs in the fragment for the track.
106                let default_dur =
107                    traf.tfhd.default_sample_duration.unwrap_or(trex.default_sample_duration);
108
109                for trun in traf.truns.iter() {
110                    info.total_sample_duration += trun.total_duration(default_dur);
111                }
112
113                info.total_sample_count = traf.total_sample_count;
114                info.traf_idx = Some(traf_idx);
115            }
116
117            seq.push(info);
118        }
119
120        MoofSegment { moof, mvex, seq }
121    }
122
123    /// Try to get the Track Fragment atom associated with the track identified by `track_num`.
124    fn try_get_traf(&self, track_num: usize) -> Option<&TrafAtom> {
125        debug_assert!(track_num < self.seq.len());
126        self.seq[track_num].traf_idx.map(|idx| &self.moof.trafs[idx])
127    }
128}
129
130impl StreamSegment for MoofSegment {
131    fn sequence_num(&self) -> u32 {
132        self.moof.mfhd.sequence_number
133    }
134
135    fn sample_timing(&self, track_num: usize, sample_num: u32) -> Result<Option<SampleTiming>> {
136        // Get the track fragment associated with track_num.
137        let traf = match self.try_get_traf(track_num) {
138            Some(traf) => traf,
139            None => return Ok(None),
140        };
141
142        let mut sample_num_rel = sample_num - self.seq[track_num].first_sample;
143        let mut trun_ts_offset = self.seq[track_num].first_ts;
144
145        let default_dur = traf
146            .tfhd
147            .default_sample_duration
148            .unwrap_or(self.mvex.trexs[track_num].default_sample_duration);
149
150        for trun in traf.truns.iter() {
151            // If the sample is contained within the this track run, get the timing of of the
152            // sample.
153            if sample_num_rel < trun.sample_count {
154                let (ts, dur) = trun.sample_timing(sample_num_rel, default_dur);
155                return Ok(Some(SampleTiming { ts: trun_ts_offset + ts, dur }));
156            }
157
158            let trun_dur = trun.total_duration(default_dur);
159
160            sample_num_rel -= trun.sample_count;
161            trun_ts_offset += trun_dur;
162        }
163
164        Ok(None)
165    }
166
167    fn ts_sample(&self, track_num: usize, ts: u64) -> Result<Option<u32>> {
168        // Get the track fragment associated with track_num.
169        let traf = match self.try_get_traf(track_num) {
170            Some(traf) => traf,
171            None => return Ok(None),
172        };
173
174        let mut sample_num = self.seq[track_num].first_sample;
175        let mut ts_accum = self.seq[track_num].first_ts;
176
177        let default_dur = traf
178            .tfhd
179            .default_sample_duration
180            .unwrap_or(self.mvex.trexs[track_num].default_sample_duration);
181
182        for trun in traf.truns.iter() {
183            // Get the total duration of this track run.
184            let trun_dur = trun.total_duration(default_dur);
185
186            // If the timestamp after the track run is greater than the desired timestamp, then the
187            // desired sample must be in this run of samples.
188            if ts_accum + trun_dur > ts {
189                sample_num += trun.ts_sample(ts - ts_accum, default_dur);
190                return Ok(Some(sample_num));
191            }
192
193            sample_num += trun.sample_count;
194            ts_accum += trun_dur;
195        }
196
197        Ok(None)
198    }

200    fn sample_data(
201        &self,
202        track_num: usize,
203        sample_num: u32,
204        get_offset: bool,
205    ) -> Result<SampleDataDesc> {
206        // Get the track fragment associated with track_num.
207        let traf = self.try_get_traf(track_num).unwrap();
208
209        // If an explicit anchor-point is set, then use that for the position, otherwise use the
210        // first-byte of the enclosing moof atom.
211        let traf_base_pos = match traf.tfhd.base_data_offset {
212            Some(pos) => pos,
213            _ => self.moof.moof_base_pos,
214        };
215
216        let mut sample_num_rel = sample_num - self.seq[track_num].first_sample;
217        let mut trun_offset = traf_base_pos;
218
219        let default_size =
220            traf.tfhd.default_sample_size.unwrap_or(self.mvex.trexs[track_num].default_sample_size);
221
222        for trun in traf.truns.iter() {
223            // If a data offset is present for this track fragment run, then calculate the new base
224            // position for the run. When a data offset is not present, do nothing because this run
225            // follows the previous run.
226            if let Some(offset) = trun.data_offset {
227                // The offset for the run is relative to the anchor-point defined in the track
228                // fragment header.
229                trun_offset = if offset.is_negative() {
230                    traf_base_pos - u64::from(offset.wrapping_abs() as u32)
231                }
232                else {
233                    traf_base_pos + offset as u64
234                };
235            }
236
237            if sample_num_rel < trun.sample_count {
238                let (offset, size) = if get_offset {
239                    // Get the size and offset of the sample.
240                    let (offset, size) = trun.sample_offset(sample_num_rel, default_size);
241                    (Some(offset), size)
242                }
243                else {
244                    // Just get the size of the sample.
245                    let size = trun.sample_size(sample_num_rel, default_size);
246                    (None, size)
247                };
248
249                return Ok(SampleDataDesc { base_pos: trun_offset, size, offset });
250            }
251
252            // Get the total size of the track fragment run.
253            let trun_size = trun.total_size(default_size);
254
255            sample_num_rel -= trun.sample_count;
256            trun_offset += trun_size;
257        }
258
259        decode_error("isomp4: invalid sample index")
260    }
261
262    fn track_sample_range(&self, track_num: usize) -> Range<u32> {
263        debug_assert!(track_num < self.seq.len());
264
265        let track = &self.seq[track_num];
266        track.first_sample..track.first_sample + track.total_sample_count
267    }
268
269    fn track_ts_range(&self, track_num: usize) -> Range<u64> {
270        debug_assert!(track_num < self.seq.len());
271
272        let track = &self.seq[track_num];
273        track.first_ts..track.first_ts + track.total_sample_duration
274    }
275}

277fn get_chunk_offset(
278    stco: &Option<StcoAtom>,
279    co64: &Option<Co64Atom>,
280    chunk: usize,
281) -> Result<Option<u64>> {
282    // Get the offset from either the stco or co64 atoms.
283    if let Some(stco) = stco.as_ref() {
284        // 32-bit offset
285        if let Some(offset) = stco.chunk_offsets.get(chunk) {
286            Ok(Some(u64::from(*offset)))
287        }
288        else {
289            decode_error("isomp4: missing stco entry")
290        }
291    }
292    else if let Some(co64) = co64.as_ref() {
293        // 64-bit offset
294        if let Some(offset) = co64.chunk_offsets.get(chunk) {
295            Ok(Some(*offset))
296        }
297        else {
298            decode_error("isomp4: missing co64 entry")
299        }
300    }
301    else {
302        // This should never happen because it is mandatory to have either a stco or co64 atom.
303        decode_error("isomp4: missing stco or co64 atom")
304    }
305}

308    moov: MoovAtom,
309}
310
311impl MoovSegment {
312    /// Instantiate a segment from the provide moov atom.
313    pub fn new(moov: MoovAtom) -> MoovSegment {
314        MoovSegment { moov }
315    }
316}
317
318impl StreamSegment for MoovSegment {
319    fn sequence_num(&self) -> u32 {
320        // The segment defined by the moov atom is always 0.
321        0
322    }
323
324    fn sample_timing(&self, track_num: usize, sample_num: u32) -> Result<Option<SampleTiming>> {
325        // Get the trak atom associated with track_num.
326        debug_assert!(track_num < self.moov.traks.len());
327
328        let trak = &self.moov.traks[track_num];
329
330        // Find the sample timing. Note, complexity of O(N).
331        let timing = trak.mdia.minf.stbl.stts.find_timing_for_sample(sample_num);
332
333        if let Some((ts, dur)) = timing {
334            Ok(Some(SampleTiming { ts, dur }))
335        }
336        else {
337            Ok(None)
338        }
339    }
340
341    fn ts_sample(&self, track_num: usize, ts: u64) -> Result<Option<u32>> {
342        // Get the trak atom associated with track_num.
343        debug_assert!(track_num < self.moov.traks.len());
344
345        let trak = &self.moov.traks[track_num];
346
347        // Find the sample timestamp. Note, complexity of O(N).
348        Ok(trak.mdia.minf.stbl.stts.find_sample_for_timestamp(ts))
349    }
350
351    fn sample_data(
352        &self,
353        track_num: usize,
354        sample_num: u32,
355        get_offset: bool,
356    ) -> Result<SampleDataDesc> {
357        // Get the trak atom associated with track_num.
358        debug_assert!(track_num < self.moov.traks.len());
359
360        let trak = &self.moov.traks[track_num];
361
362        // Get the constituent tables.
363        let stsz = &trak.mdia.minf.stbl.stsz;
364        let stsc = &trak.mdia.minf.stbl.stsc;
365        let stco = &trak.mdia.minf.stbl.stco;
366        let co64 = &trak.mdia.minf.stbl.co64;
367
368        // Find the sample-to-chunk mapping. Note, complexity of O(log N).
369        let group = stsc
370            .find_entry_for_sample(sample_num)
371            .ok_or(Error::DecodeError("invalid sample index"))?;
372
373        // Index of the sample relative to the chunk group.
374        let sample_in_group = sample_num - group.first_sample;
375
376        // Index of the chunk containing the sample relative to the chunk group.
377        let chunk_in_group = sample_in_group / group.samples_per_chunk;
378
379        // Index of the chunk containing the sample relative to the entire stream.
380        let chunk_in_stream = group.first_chunk + chunk_in_group;
381
382        // Get the byte position of the first sample of the chunk containing the sample.
383        let base_pos = get_chunk_offset(stco, co64, chunk_in_stream as usize)?.unwrap();
384
385        // Determine the absolute sample byte position if requested by calculating the offset of
386        // the sample from the base position of the chunk.
387        let offset = if get_offset {
388            // Index of the sample relative to the chunk containing the sample.
389            let sample_in_chunk = sample_in_group - (chunk_in_group * group.samples_per_chunk);
390
391            // Calculat the byte offset of the sample relative to the chunk containing it.
392            let offset = match stsz.sample_sizes {
393                SampleSize::Constant(size) => {
394                    // Constant size samples can be calculated directly.
395                    u64::from(sample_in_chunk) * u64::from(size)
396                }
397                SampleSize::Variable(ref entries) => {
398                    // For variable size samples, sum the sizes of all the samples preceeding the
399                    // desired sample in the chunk.
400                    let chunk_first_sample = (sample_num - sample_in_chunk) as usize;
401
402                    if let Some(samples) = entries.get(chunk_first_sample..sample_num as usize) {
403                        samples.iter().map(|&size| u64::from(size)).sum()
404                    }
405                    else {
406                        return decode_error("isomp4: missing one or more stsz entries");
407                    }
408                }
409            };
410
411            Some(offset)
412        }
413        else {
414            None
415        };
416
417        // Get the size in bytes of the sample.
418        let size = match stsz.sample_sizes {
419            SampleSize::Constant(size) => size,
420            SampleSize::Variable(ref entries) => {
421                if let Some(size) = entries.get(sample_num as usize) {
422                    *size
423                }
424                else {
425                    return decode_error("isomp4: missing stsz entry");
426                }
427            }
428        };
429
430        Ok(SampleDataDesc { base_pos, size, offset })
431    }
432
433    fn track_sample_range(&self, track_num: usize) -> Range<u32> {
434        debug_assert!(track_num < self.moov.traks.len());
435
436        0..self.moov.traks[track_num].mdia.minf.stbl.stsz.sample_count
437    }
438
439    fn track_ts_range(&self, track_num: usize) -> Range<u64> {
440        debug_assert!(track_num < self.moov.traks.len());
441
442        0..self.moov.traks[track_num].mdia.minf.stbl.stts.total_duration
443    }
