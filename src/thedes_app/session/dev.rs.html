76    pub async fn run(
77        &mut self,
78        app: &mut App,
79        game: &mut Game,
80    ) -> Result<(), Error> {
81        while self.handle_input(app, game).await? {
82            self.render(app)?;
83            tokio::select! {
84                _ = app.tick_session.tick() => (),
85                _ = app.cancel_token.cancelled() => Err(Error::Cancelled)?,
86            }
87        }
88        Ok(())
89    }

92        app.canvas

118    async fn handle_input(
119        &mut self,
120        app: &mut App,
121        game: &mut Game,
122    ) -> Result<bool, Error> {
123        let Ok(mut events) = app.events.read_until_now() else {
124            Err(Error::Cancelled)?
125        };
126
127        let mut should_continue = true;
128        let mut hit = false;
129        while let Some(event) = events.next().filter(|_| should_continue) {
130            let Event::Key(key) = event else { continue };
131
132            let command = self.key_bindings.command_for(key).cloned();
133
134            should_continue = match command {
135                Some(command) => {
136                    hit = true;
137                    self.run_command(game, command).await?
138                },
139                None => match key {
140                    KeyEvent {
141                        main_key: Key::Char(ch),
142                        ctrl: false,
143                        alt: false,
144                        shift: false,
145                    } => {
146                        hit = true;
147                        self.run_command(game, Command::Run(ch)).await?
148                    },
149                    _ => true,
150                },
151            };
152        }
153        Ok(!hit)
154    }
155
156    async fn run_command(
157        &mut self,
158        game: &mut Game,
159        command: Command,
160    ) -> Result<bool, Error> {
161        match command {
162            Command::Run(ch) => ScriptTable::run_reading(ch, game).await?,
163            Command::RunPrevious => {
164                ScriptTable::run_reading(self.prev, game).await?
165            },
166            Command::Exit => return Ok(false),
167        }
168        Ok(true)
169    }
