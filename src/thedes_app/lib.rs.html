59impl fmt::Display for MainMenuItem {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        f.write_str(match self {
62            Self::NewGame => "New Game",
63            Self::LoadGame => "Load Game",
64            Self::Settings => "Settings",
65            Self::Quit => "Quit",
66        })
67    }
68}

70pub async fn root(mut app: thedes_tui::core::App) -> Result<(), Error> {
71    let main_menu_items = [
72        MainMenuItem::NewGame,
73        MainMenuItem::LoadGame,
74        MainMenuItem::Settings,
75        MainMenuItem::Quit,
76    ];
77
78    let quit_position = main_menu_items
79        .iter()
80        .position(|item| *item == MainMenuItem::Quit)
81        .unwrap_or_default();
82
83    let main_menu_bindings = menu::default_key_bindings()
84        .with(Key::Char('q'), menu::Command::SelectConfirm(quit_position));
85
86    let mut main_menu = Menu::new("=== T H E D E S ===", main_menu_items)?
87        .with_keybindings(main_menu_bindings);
88
89    let mut seed_input = Input::new(input::Config {
90        max: 8,
91        filter: |ch: char| ch.is_ascii_hexdigit(),
92        title: "New World Seed",
93    })?;
94
95    let mut empty_seed_info = Info::new("Error!", "Seed cannot be empty");
96
97    loop {
98        main_menu.run(&mut app).await?;
99
100        match main_menu.output() {
101            MainMenuItem::NewGame => {
102                seed_input.run(&mut app).await?;
103                let seed_str = seed_input.output();
104                if seed_str.is_empty() {
105                    empty_seed_info.run(&mut app).await?;
106                } else {
107                    u32::from_str_radix(&seed_str, 16)?;
108                }
109            },
110            MainMenuItem::LoadGame => {},
111            MainMenuItem::Settings => {},
112            MainMenuItem::Quit => break,
113        }
114    }
115
116    Ok(())
117}
