11#[derive(Debug, Error)]

27pub async fn root(mut app: thedes_tui::core::App) -> Result<(), Error> {
28    let mut timer = app.timer.new_participant();
29
30    let min = CoordPair { y: 1, x: 0 };
31    let max = app.canvas.size() - 1;
32    let mut curr = min + (min + max) / 2;
33
34    'main: loop {
35        app.canvas
36            .queue([screen::Command::ClearScreen(BasicColor::Black.into())]);
37        thedes_tui::text::inline(
38            &mut app,
39            CoordPair { y: 0, x: 0 },
40            "Hello, World! Q/Esc to quit, HJKL to move",
41            ColorPair {
42                foreground: BasicColor::Black.into(),
43                background: BasicColor::LightGreen.into(),
44            },
45        )?;
46        app.canvas.queue([screen::Command::new_mutation(
47            curr,
48            MutateGrapheme(Set('O'.into())).then(MutateColors(Set(
49                ColorPair {
50                    foreground: BasicColor::White.into(),
51                    background: BasicColor::Black.into(),
52                },
53            ))),
54        )]);
55        if app.canvas.flush().is_err() {
56            tracing::info!("Screen command receiver disconnected");
57            break;
58        }
59
60        let Ok(events) = app.events.read_until_now() else {
61            tracing::info!("Event sender disconnected");
62            break;
63        };
64        for event in events {
65            let Event::Key(KeyEvent {
66                alt: false,
67                ctrl: false,
68                shift: false,
69                main_key,
70            }) = event
71            else {
72                continue;
73            };
74
75            match main_key {
76                Key::Char('q') | Key::Char('Q') | Key::Esc => break 'main,
77                Key::Char('h') | Key::Char('H') => {
78                    curr.x = curr.x.saturating_sub(1).max(min.x).min(max.x);
79                },
80                Key::Char('j') | Key::Char('J') => {
81                    curr.y = curr.y.saturating_add(1).max(min.y).min(max.y);
82                },
83                Key::Char('k') | Key::Char('K') => {
84                    curr.y = curr.y.saturating_sub(1).max(min.y).min(max.y);
85                },
86                Key::Char('l') | Key::Char('L') => {
87                    curr.x = curr.x.saturating_add(1).max(min.x).min(max.x);
88                },
89                _ => (),
90            }
91        }
92
93        timer.tick().await;
94    }
95
96    Ok(())
97}
