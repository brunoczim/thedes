10#[derive(Debug, Error)]
11pub enum Error {
12    #[error("Failed to render TUI")]
13    RenderText(
14        #[from]
15        #[source]
16        thedes_tui::text::Error,
17    ),
18    #[error("Failed to interact with screen canvas")]
19    CanvasFlush(
20        #[from]
21        #[source]
22        thedes_tui::core::screen::FlushError,
23    ),
24    #[error("Failed to run main menu")]
25    MainMenu(
26        #[from]
27        #[source]
28        menu::Error,
29    ),
30    #[error("Failed to run input")]
31    SeedInput(
32        #[from]
33        #[source]
34        input::Error,
35    ),
36}

38#[derive(Debug, Clone, Copy, PartialEq, Eq)]
39enum MainMenuItem {
40    NewGame,
41    LoadGame,
42    Settings,
43    Quit,
44}
45
46impl fmt::Display for MainMenuItem {
47    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
48        f.write_str(match self {
49            Self::NewGame => "New Game",
50            Self::LoadGame => "Load Game",
51            Self::Settings => "Settings",
52            Self::Quit => "Quit",
53        })
54    }
55}
56
57pub async fn root(mut app: thedes_tui::core::App) -> Result<(), Error> {
58    let main_menu_items = [
59        MainMenuItem::NewGame,
60        MainMenuItem::LoadGame,
61        MainMenuItem::Settings,
62        MainMenuItem::Quit,
63    ];
64
65    let quit_position = main_menu_items
66        .iter()
67        .position(|item| *item == MainMenuItem::Quit)
68        .unwrap_or_default();
69
70    let main_menu_bindings = menu::default_key_bindings()
71        .with(Key::Char('q'), menu::Command::SelectConfirm(quit_position));
72
73    let mut main_menu = Menu::new("=== T H E D E S ===", &main_menu_items)?
74        .with_keybindings(main_menu_bindings);
75
76    let mut seed_input = Input::new(input::Config {
77        max: 32,
78        filter: |ch: char| ch.is_ascii_hexdigit(),
79        title: "New World Seed",
80    })?;
81
82    loop {
83        main_menu.run(&mut app).await?;
84
85        match main_menu.output() {
86            MainMenuItem::NewGame => {
87                seed_input.run(&mut app).await?;
88            },
89            MainMenuItem::LoadGame => {},
90            MainMenuItem::Settings => {},
91            MainMenuItem::Quit => break,
92        }
93    }
94
95    Ok(())
96}
