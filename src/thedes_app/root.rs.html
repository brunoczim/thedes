84#[derive(Debug, Clone, Copy, PartialEq, Eq)]
85enum MainMenuItem {
86    NewGame,
87    LoadGame,
88    Settings,
89    Quit,
90}
91
92impl fmt::Display for MainMenuItem {
93    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
94        f.write_str(match self {
95            Self::NewGame => "New Game",
96            Self::LoadGame => "Load Game",
97            Self::Settings => "Settings",
98            Self::Quit => "Quit",
99        })
100    }
101}

103#[derive(Debug, Clone)]
104pub struct Component {
105    main_menu: Menu<MainMenuItem>,
106    new_game: new_game::Component,
107    session_config: session::Config,
108}
109
110impl Component {
111    pub fn new() -> Result<Self, InitError> {
112        let main_menu_items = [
113            MainMenuItem::NewGame,
114            MainMenuItem::LoadGame,
115            MainMenuItem::Settings,
116            MainMenuItem::Quit,
117        ];
118
119        let quit_position = main_menu_items
120            .iter()
121            .position(|item| *item == MainMenuItem::Quit)
122            .ok_or(InitError::MissingQuit)?;
123
124        let main_menu_bindings = menu::default_key_bindings()
125            .with(Key::Char('q'), menu::Command::SelectConfirm(quit_position));
126
127        let main_menu = Menu::new("=== T H E D E S ===", main_menu_items)?
128            .with_keybindings(main_menu_bindings);
129
130        let new_game = new_game::Component::new()?;
131
132        Ok(Self { main_menu, new_game, session_config: session::Config::new() })
133    }
134
135    pub async fn run(
136        &mut self,
137        app: &mut thedes_tui::core::App,
138    ) -> Result<(), RunError> {
139        loop {
140            self.main_menu.run(app).await?;
141
142            match self.main_menu.output() {
143                MainMenuItem::NewGame => {
144                    self.new_game.run(app).await?;
145                    let map = Map::new(Rect {
146                        top_left: CoordPair { x: 0, y: 0 },
147                        size: CoordPair { x: 1000, y: 1000 },
148                    })?;
149                    let player_position = PlayerPosition::new(
150                        CoordPair { y: 50, x: 50 },
151                        Direction::Up,
152                    )?;
153                    let game = Game::new(map, player_position)?;
154                    let mut session =
155                        self.session_config.clone().finish(game)?;
156                    session.run(app).await?;
157                },
158
159                MainMenuItem::LoadGame => {},
160                MainMenuItem::Settings => {},
161                MainMenuItem::Quit => break,
162            }
163        }
164
165        Ok(())
166    }
167}
